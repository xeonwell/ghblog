一个更新OE地址簿的批处理，部分适用

适合我们公司内部情况而写的一个批处理，使用时有什么问题可以与我联系，中间用到了pskill和reg两个文件，可以上网搜索

@echo off
@if {%os%}=={Windows_NT} goto winnt

:win98
rem win98
@echo "win98操作系统不能应用本更新程序"
@pause
@exit

:winnt
@pause
@if exist "%TEMP%\%DATE%.wab" goto enddo
@pskill wab >nul
@pskill msimn >nul
@copy /y "%USERPROFILE%\Application Data\Microsoft\Address Book\%USERNAME%.wab" "%TEMP%\%DATE%.wab" >nul
@copy /y addrbk.wab "%USERPROFILE%\Application Data\Microsoft\Address Book\%USERNAME%.wab" >nul
@reg add "HKEY_CURRENT_USER\Software\Microsoft\WAB\WAB4\Wab File Name" /ve /t REG_SZ /d "%USERPROFILE%\Application Data\Microsoft\Address Book\%USERNAME%.wab" /f >nul

:enddo
@rem del the tmp files
@del reg.exe >nul
@del pskill.exe >nul
@del addrbk.wab >nul
@del auto_update_addr_book.bat >nul
@exit
@pause
